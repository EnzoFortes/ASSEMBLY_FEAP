SYS_EXIT        equ 1
SYS_READ        equ 3
SYS_WRITE       equ 4
STDIN           equ 0
STDOUT          equ 1


section .data             ;seção de dados iniciada

    msg1 db "Digite seu nome: "
    len1 equ $-msg1
    
    msg2 db "Saudações, "
    len2 equ $-msg2

section .bss                                ; Seção de declaração de vaviaveis a serem processadas
    
    nome resb 10

section .text                                   ; Seção de exeção
    global _start       

_start:                                         ; Iniciando programação

    ; Pede para o usuário colocar o seu nome
    mov eax, SYS_WRITE
    mov ebx, STDOUT
    mov ecx, msg1
    mov edx, len1
    int 0x80

    ; Lê o nome informado pelo usuário
    mov eax, SYS_READ
    mov ebx, STDIN
    mov ecx, nome
    mov edx, 10
    int 0x80
    
xor esi, esi
.loop:
    mov edi, [nome + esi]
    cmp edi, 0
    je .end
    inc esi
    jmp .loop

.end:
    ; Imprime a saudação
    mov eax, SYS_WRITE
    mov ebx, STDOUT
    mov ecx, msg2
    mov edx, len2
    int 0x80

    ; Imprime o nome do usuário
    mov eax, SYS_WRITE
    mov ebx, STDOUT
    mov ecx, nome
    mov edx, esi
    int 0x80

exit:
    mov eax, SYS_EXIT
    xor ebx, ebx
    int 0x80
